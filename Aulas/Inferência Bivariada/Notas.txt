3. Distribuição normal
tapply(X=dados_Hair_etal$use, INDEX=dados_Hair_etal$size, FUN=shapiro.test)

tapply(X=dados_Hair_etal$use, INDEX=dados_Hair_etal$size, FUN=var)

Res.anova<-aov(use~size,dados_Hair_etal)
